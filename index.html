// โค้ดสำหรับ Google Apps Script (Code.gs) // อัปเดตเพื่อรองรับข้อมูลที่ละเอียดขึ้นจากระบบตะกร้าสินค้า function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Orders"); var data = JSON.parse(e.postData.contents); // 1. จัดการรูปสลิป var slipUrl = "ไม่มีแนบมา"; if (data.slipData) { // ** อย่าลืมเปลี่ยน Folder ID เป็นของคุณเองนะครับ ** var folderId = "13DXUzYtZr6B5E5se0PbFxoGjRdwSdi-Z"; var folder = DriveApp.getFolderById(folderId); var fileName = "Slip_" + data.receiverName + "_" + new Date().getTime() + ".png"; var blob = Utilities.newBlob(Utilities.base64Decode(data.slipData), "image/png", fileName); var file = folder.createFile(blob); slipUrl = file.getUrl(); } // 2. บันทึกข้อมูลลงใน Sheet ตามลำดับที่คุณจัดไว้ // ลำดับ: 1.วันที่ | 2.ชื่อ LINE | 3.ชื่อผู้รับ | 4.รายการสินค้า | 5.ยอดรวม | 6.สถานที่รับ | 7.เวลาที่รับ | 8.ข้อมูลติดต่อ | 9.สถานะ | 10.ลิงก์สลิป sheet.appendRow([ new Date(), data.userName, data.receiverName, data.items, // จะเป็นข้อความเช่น "บราวนี่ x2, น้ำส้ม x1" data.amount, // ยอดรวมราคาสุทธิ data.location, data.time, data.contact, "รอตรวจสอบ", slipUrl ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "slipUrl": slipUrl })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.message })).setMimeType(ContentService.MimeType.JSON); } } // ฟังก์ชันสำหรับขอสิทธิ์เข้าถึง (รันครั้งเดียวตอนแก้โค้ดใหม่) function triggerPermission() { DriveApp.getStorageUsed(); DriveApp.createFile("test.txt", "hello"); }
